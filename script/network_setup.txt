#!/bin/bash

# Define log file path
LOGFILE="../logs/v0icex.log"

# Ensure the log directory exists
mkdir -p ../logs

# Create the log file if it doesn't exist
touch $LOGFILE

# Start Tor
echo "[+] Starting Tor..." | tee -a $LOGFILE
sudo service tor start || {
    echo "[!] Failed to start Tor." | tee -a $LOGFILE
    exit 1
}

# Prompt for VPN configuration file path
echo "[+] Enter the path to your VPN configuration file:"
read VPN_CONFIG

# Start VPN
echo "[+] Starting VPN with configuration: $VPN_CONFIG..." | tee -a
